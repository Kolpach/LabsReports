#define show(k,n) for(int i=0; i<n; i++) cout<<k[i]<<' ';cout<<endl
#include <iostream>
using namespace std;
int main()
{
	setlocale(0, "");
	int n;
	cout << "¬ведите количество элементов" << endl;
	cin >> n;
	int *a = new int[n];
	int *b = new int[n];
	int min, max;
	cout << "¬ведите n элементов:" << endl;
	cin >> a[0];
	b[0] = -1;
	min = a[0];
	max = a[0];
	for (int i = 1; i < n; i++)//заполнение массива и вычисление max и min
	{
		cin >> a[i];
		b[i] = -1;
		if (a[i] > max)
			max = a[i];
		else if (a[i] < min)
			min = a[i];
	}

	for (int i = 0, j; i < n; i++)//добавление элементов(предполога€, что больша€ часть из них линейные)
	{
		j = (a[i] - min) *(n - 1) / (max - min);
		if (a[j] < a[i])
		{
			int y = a[j];
			a[j] = a[i];
			a[i] = y;
		}
	}
	show(a, n);

	bool end = true;
	while (end)
	{
		end = false;
		for (int i = 1; i < n; i++)
		{
			if (a[i] < a[i - 1])
			{
				int y = a[i - 1];
				a[i - 1] = a[i];
				a[i] = y;
				end = true;
			}
		}
	}
	show(a, n);

	system("pause");
	delete[] a;
	delete[] b;
	return 0;
}
