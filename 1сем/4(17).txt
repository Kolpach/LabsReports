int findfree(int* A, int n)
{
	int i = n;
	while (A[i] != 0 && i < 10)
		i++;
	if (i > 9)
		return -1;
	return i;
}
bool checkMas(int **A, int **B, int **C)
{
	int a = 0, b = 0, c = 0;
	for (int i = 0; i < 4; i++)
	{
		a *= 10;
		a = a + *A[i];
	}
	for (int i = 0; i < 5; i++)
	{
		b *= 10;
		b = b + *B[i];
	}
	for (int i = 0; i < 6; i++)
	{
		c *= 10;
		c = c + *C[i];
	}

	if (a + b == c)
		return true;
	else
		return false;
}
