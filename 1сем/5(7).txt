#include <iostream>
#include <cmath>
using namespace std;
double row(double x, double eps, int &n)
{
	double s, sn;
	for (s = 0, sn = x, n = 1; fabs(sn) > eps; n++)

	{
		s += sn;

		sn = -sn * x / n;
	}
	return s;
}
