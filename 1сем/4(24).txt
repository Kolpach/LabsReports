#include <iostream>
using namespace std;
int main()
{
	setlocale(0, "");
	int month[12] = { 31,28,31,30,31,30,31,31,30,31,30,31 };
	int count = 0, lCount, max=0, mYear=0;
	for (int i =1; i <= 100; i++)
	{
		lCount = 0;
		if (i % 4 == 0)
			month[1] = 29;
		else
			month[1] = 28;
		for (int j = 1; j <= 12; j++)
			if (i % j == 0 && month[j-1] >= i / j && month[j - 1] >= 1)
			{
				count++;
				lCount++;
			}
		if (max < lCount)
		{
			max = lCount;
			mYear = i;
		}
	}
	cout << "Количество появлений \"свойства\": " << count << endl;
	if(mYear!=100)
		cout << "Год с наибольшим количеством \"свойств\" - 19" << mYear << ". Их количество: " << max << endl;
	else
		cout << "Год с наибольшим количеством \"свойств\" - 2000. Их количество: " << max << endl;
	system("pause");
	return 0;
}
